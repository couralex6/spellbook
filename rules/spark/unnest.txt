Rule: when explode is used in the query, replace explode with unnest
SELECT posexplode(array) FROM table -> SELECT vals.val FROM table1, unnest(arrayFromTable1) as vals(val)
Example:
select explode(sequence(to_date('2022-09-01'), now(), interval 1 day)) as day
->
with
    day_seq as(
        SELECT
      (
        sequence(
          cast('2022-09-01' as date),
          cast(now() as date),
          interval '1' day
        )
      ) day
  )
select
  days.day
from
  day_seq
  cross join unnest(day) as days(day)
